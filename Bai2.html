🐍 Python là gì?

Python là ngôn ngữ lập trình bậc cao, được thiết kế với triết lý nhấn mạnh tính dễ đọc của code. Python được tạo ra bởi Guido van Rossum và phát hành lần đầu năm 1991.

💡 Đặc điểm nổi bật:
  • Cú pháp đơn giản, dễ học
  • Thụt lề (indentation) thay vì dấu ngoặc
  • Kiểu dữ liệu động (dynamic typing)
  • Thư viện phong phú

📐 Tại sao thụt lề quan trọng?

Khác với các ngôn ngữ như C++, Java dùng dấu {}, Python dùng thụt lề để xác định khối lệnh. Mỗi khối thường thụt vào 4 dấu cách hoặc 1 tab.

❌ Sai - Lỗi IndentationError

if True:
print("Thiếu thụt lề!")  # Lỗi!

for i in range(5):
print(i)  # Lỗi!

✅ Đúng - Code hợp lệ

if True:
    print("Thụt lề đúng!")  # OK

for i in range(5):
    print(i)  # OK
Ví dụ thụt lề nhiều cấp 
def tinh_tong(danh_sach):
    """Tính tổng các số dương"""
    tong = 0                        # Cấp 1: trong hàm
    for so in danh_sach:
        if so > 0:                   # Cấp 2: trong for
            tong += so               # Cấp 3: trong if
    return tong

# Gọi hàm (không thụt lề)
ket_qua = tinh_tong([1, -2, 3, -4, 5])
print(ket_qua)  # Output: 9
🎯 Ví dụ thực tế: Hệ thống chấm điểm học sinh
# File: cham_diem.py
# Chương trình xếp loại học lực

def xep_loai_hoc_luc(diem):
    """Xếp loại học lực dựa trên điểm"""
    # Cấp 1: Trong hàm
    if diem >= 9.0:
        # Cấp 2: Trong if
        xep_loai = "Xuất sắc"
        hoc_bong = 5000000
        if diem == 10.0:
            # Cấp 3: Trong if lồng nhau
            hoc_bong = 10000000
            print("🎉 Chúc mừng! Điểm tuyệt đối!")
    elif diem >= 8.0:
        xep_loai = "Giỏi"
        hoc_bong = 3000000
    elif diem >= 6.5:
        xep_loai = "Khá"
        hoc_bong = 1000000
    elif diem >= 5.0:
        xep_loai = "Trung bình"
        hoc_bong = 0
    else:
        xep_loai = "Yếu"
        hoc_bong = 0
        print("⚠️ Cần cố gắng hơn!")
    
    return xep_loai, hoc_bong

# Sử dụng hàm (không thụt lề - ngoài hàm)
ten_hoc_sinh = "Trần Thị Bình"
diem_tb = 8.7

loai, tien_thuong = xep_loai_hoc_luc(diem_tb)
print(f"\n👤 Học sinh: {ten_hoc_sinh}")
print(f"📊 Điểm TB: {diem_tb}")
print(f"🏆 Xếp loại: {loai}")
print(f"💰 Học bổng: {tien_thuong:,} VNĐ")

# OUTPUT:
# 👤 Học sinh: Trần Thị Bình
# 📊 Điểm TB: 8.7
# 🏆 Xếp loại: Giỏi
# 💰 Học bổng: 3,000,000 VNĐ
💡 Phân tích thụt lề:
  • • Cấp 0 (không thụt): Định nghĩa hàm, biến toàn cục
  • • Cấp 1 (4 space): Nội dung trong hàm, docstring
  • • Cấp 2 (8 space): Khối lệnh trong if/elif/else
  • • Cấp 3 (12 space): Khối lệnh trong if lồng nhau
  • • Tất cả câu lệnh cùng cấp PHẢI có cùng số dấu cách thụt lề
📦 Module là gì?

Module là file Python chứa code có thể tái sử dụng. Python có hàng nghìn module có sẵn, và bạn cũng có thể tự tạo module của riêng mình.

# Cách 1: Import toàn bộ module
import math
print(math.pi)        # 3.141592653589793
print(math.sqrt(16))  # 4.0

# Cách 2: Import với alias (bí danh)
import datetime as dt

# Cách 3: Import thành phần cụ thể
from datetime import datetime, timedelta
from math import pi, sqrt

# Cách 4: Import tất cả (không khuyến khích)
from os import *  # Tránh dùng cách này!

# File: time_manager.py
# Ứng dụng quản lý và tính toán thời gian

# Import các module cần thiết
from datetime import datetime, timedelta
import math

def tinh_ngay_deadline(ngay_bat_dau, so_ngay_lam):
    """Tính ngày deadline của dự án"""
    ngay_ket_thuc = ngay_bat_dau + timedelta(days=so_ngay_lam)
    return ngay_ket_thuc

def tinh_gio_lam_viec(so_ngay, gio_moi_ngay):
    """Tính tổng số giờ làm việc"""
    tong_gio = so_ngay * gio_moi_ngay
    tong_phut = tong_gio * 60
    return tong_gio, tong_phut

def tinh_khoang_cach_ngay(ngay_1, ngay_2):
    """Tính khoảng cách giữa 2 ngày"""
    delta = ngay_2 - ngay_1
    return delta.days

# Sử dụng các hàm
print("========== QUẢN LÝ DỰ ÁN ==========")

# Thông tin dự án
ten_du_an = "Website Bán Hàng"
ngay_bd = datetime(2024, 1, 15)
so_ngay_du_kien = 45
gio_lam_moi_ngay = 8

# Tính toán
ngay_kt = tinh_ngay_deadline(ngay_bd, so_ngay_du_kien)
tong_gio, tong_phut = tinh_gio_lam_viec(so_ngay_du_kien, gio_lam_moi_ngay)
ngay_con_lai = tinh_khoang_cach_ngay(datetime.now(), ngay_kt)

# In kết quả
print(f"📋 Dự án: {ten_du_an}")
print(f"📅 Ngày bắt đầu: {ngay_bd.strftime('%d/%m/%Y')}")
print(f"🏁 Ngày kết thúc: {ngay_kt.strftime('%d/%m/%Y')}")
print(f"⏱️  Tổng thời gian: {so_ngay_du_kien} ngày")
print(f"⏰ Giờ làm việc: {tong_gio} giờ ({tong_phut:,} phút)")
print(f"⏳ Còn lại: {ngay_con_lai} ngày")

# Sử dụng math để tính tuần
so_tuan = math.ceil(so_ngay_du_kien / 7)
print(f"📆 Tương đương: ~{so_tuan} tuần")
print("==================================")

# OUTPUT:
# ========== QUẢN LÝ DỰ ÁN ==========
# 📋 Dự án: Website Bán Hàng
# 📅 Ngày bắt đầu: 15/01/2024
# 🏁 Ngày kết thúc: 29/02/2024
# ⏱️  Tổng thời gian: 45 ngày
# ⏰ Giờ làm việc: 360 giờ (21,600 phút)
# ⏳ Còn lại: -299 ngày
# 📆 Tương đương: ~7 tuần
# ==================================
💡 Giải thích Import:
  • • datetime: Module xử lý ngày tháng, thời gian
  • • timedelta: Tính toán khoảng cách thời gian
  • • math.ceil(): Làm tròn lên số nguyên gần nhất
  • • strftime(): Format ngày tháng theo định dạng
  • • Import giúp tái sử dụng code mà không cần viết lại

⚠️ Best Practice: Đặt tất cả câu lệnh import ở đầu file, theo thứ tự: thư viện chuẩn → thư viện bên thứ 3 → module tự tạo.

🔧 Cấu trúc hàm

Hàm giúp tổ chức code thành các khối logic có thể tái sử dụng. Mỗi hàm nên làm MỘT việc cụ thể và làm tốt việc đó.

# Hàm cơ bản
def chao_mung():
    """Hàm in lời chào"""
    print("Xin chào!")

# Hàm có tham số
def chao(ten):
    """Chào một người cụ thể"""
    print(f"Xin chào, {ten}!")

# Hàm có giá trị mặc định
def luy_thua(co_so, so_mu=2):
    """Tính lũy thừa, mặc định là bình phương"""
    return co_so ** so_mu

# Hàm trả về nhiều giá trị
def thong_ke(danh_sach):
    """Trả về min, max, trung bình"""
    return min(danh_sach), max(danh_sach), sum(danh_sach)/len(danh_sach)

# Sử dụng
chao("Python")
print(luy_thua(3))       # 9 (3²)
print(luy_thua(2, 10))   # 1024 (2¹⁰)

nho, lon, tb = thong_ke([1, 2, 3, 4, 5])

# File: health_calculator.py
# Chương trình tính BMI và đánh giá sức khỏe

def tinh_bmi(can_nang, chieu_cao):
    """
    Tính chỉ số BMI (Body Mass Index)
    can_nang: kg, chieu_cao: m
    """
    bmi = can_nang / (chieu_cao ** 2)
    return round(bmi, 2)

def danh_gia_bmi(bmi):
    """Đánh giá tình trạng sức khỏe dựa trên BMI"""
    if bmi < 18.5:
        return "Thiếu cân", "⚠️", "Cần tăng cân"
    elif 18.5 <= bmi < 24.9:
        return "Bình thường", "✅", "Tuyệt vời!"
    elif 25 <= bmi < 29.9:
        return "Thừa cân", "⚠️", "Nên giảm cân"
    else:
        return "Béo phì", "❌", "Cần giảm cân gấp"

def can_nang_ly_tuong(chieu_cao, bmi_muc_tieu=22):
    """Tính cân nặng lý tưởng (mặc định BMI=22)"""
    can_nang_ideal = bmi_muc_tieu * (chieu_cao ** 2)
    return round(can_nang_ideal, 1)

def tinh_luong_nuoc(can_nang):
    """Tính lượng nước cần uống/ngày (30ml/kg)"""
    luong_nuoc_ml = can_nang * 30
    luong_nuoc_lit = luong_nuoc_ml / 1000
    return luong_nuoc_ml, luong_nuoc_lit

def hien_thi_bao_cao(ten, can_nang, chieu_cao):
    """Hiển thị báo cáo sức khỏe đầy đủ"""
    print("\n╔══════════════════════════════════════╗")
    print("║     BÁO CÁO ĐÁNH GIÁ SỨC KHỎE      ║")
    print("╚══════════════════════════════════════╝\n")
    
    # Tính toán các chỉ số
    bmi = tinh_bmi(can_nang, chieu_cao)
    tinh_trang, bieu_tuong, loi_khuyen = danh_gia_bmi(bmi)
    cn_ideal = can_nang_ly_tuong(chieu_cao)
    chenh_lech = can_nang - cn_ideal
    ml, lit = tinh_luong_nuoc(can_nang)
    
    # Hiển thị thông tin
    print(f"👤 Họ tên: {ten}")
    print(f"⚖️  Cân nặng: {can_nang} kg")
    print(f"📏 Chiều cao: {chieu_cao} m")
    print(f"\n📊 Chỉ số BMI: {bmi}")
    print(f"{bieu_tuong} Tình trạng: {tinh_trang}")
    print(f"💬 Nhận xét: {loi_khuyen}")
    
    print(f"\n🎯 Cân nặng lý tưởng: {cn_ideal} kg")
    if chenh_lech > 0:
        print(f"📉 Cần giảm: {abs(chenh_lech):.1f} kg")
    elif chenh_lech < 0:
        print(f"📈 Cần tăng: {abs(chenh_lech):.1f} kg")
    else:
        print("✨ Cân nặng lý tưởng!")
    
    print(f"\n💧 Nước cần uống: {lit:.1f}L/ngày ({ml:.0f}ml)")
    print("\n═══════════════════════════════════════\n")

# Sử dụng chương trình
hien_thi_bao_cao("Lê Văn Cường", 75, 1.75)

# OUTPUT:
# ╔══════════════════════════════════════╗
# ║     BÁO CÁO ĐÁNH GIÁ SỨC KHỎE      ║
# ╚══════════════════════════════════════╝
# 👤 Họ tên: Lê Văn Cường
# ⚖️  Cân nặng: 75 kg
# 📏 Chiều cao: 1.75 m
# 📊 Chỉ số BMI: 24.49
# ✅ Tình trạng: Bình thường
# 💬 Nhận xét: Tuyệt vời!
# 🎯 Cân nặng lý tưởng: 67.4 kg
# 📉 Cần giảm: 7.6 kg
# 💧 Nước cần uống: 2.2L/ngày (2250ml)
💡 Phân tích cấu trúc hàm:
  • • Docstring: Mô tả chức năng của hàm
  • • Tham số: Dữ liệu đầu vào cho hàm
  • • Return: Trả về kết quả (có thể nhiều giá trị)
  • • Giá trị mặc định: bmi_muc_tieu=22 nếu không truyền
  • • Mỗi hàm làm 1 việc cụ thể → dễ test, dễ maintain
